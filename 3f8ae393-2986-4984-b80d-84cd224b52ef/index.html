Scritps/get_poems.j$1

```js
/**

  • 获取网络数据，所以同步函数
  • 没有参数
  • 其实就是套用前面说过的获取 JSON 数据 Api 的代码
    /
    async function get_poems () {
    const response = await fetch(“https://v1.jinrishici.com/all.json")
    const data = await response.json()
    /     根据标点做一下换行，对作者和作品名加一些必要的格式 /
    const result = data.content.replace(/([，。；？！])/g, ‘$1 \n’)
    + ‘\n\n—— ‘+data.author+’ \n’
    + ‘《’+data.origin+’》’
    /     输出的时候整理为 Markdown 的引用格式 */
    return result.replace(/\n{3,}/g, ‘\n\n’).split(/\n/g).map(e=>’> ‘+e+’\n’).join(‘’)
    }
    module.exports = get_poems;
